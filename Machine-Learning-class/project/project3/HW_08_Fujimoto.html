tr_acc = []
va_acc = []
exp_list = np.linspace(-3, 3, 20)

for k in exp_list:
    temp_mod = LogisticRegression(solver='lbfgs', C=10**k, multi_class='auto',max_iter=1000 )
    temp_mod.fit(X_train, y_train)

plt.ylabel('Accuracy')
plt.legend()
plt.show()
